    int maxAbsoluteSum(vector<int>& nums) {
        int n=nums.size();
        int currSum=0,x=0,y=0,res=INT_MIN;//y for positive ans x for negtive
        for(int i=0;i<n;i++){
            currSum+=nums[i];
            if(currSum<0){
                res=max(res,abs(currSum-y));
                x=min(currSum,x);
            }
            else{
                res=max(res,currSum-x);
                y=max(currSum,y);
            }
        }
        return res;
        
    }
